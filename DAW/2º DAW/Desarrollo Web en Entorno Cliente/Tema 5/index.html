Manual de Validación de Formularios en JavaScript

Este manual proporciona ejemplos de cómo validar diferentes tipos de campos en un formulario utilizando JavaScript. Los mensajes de error se muestran en ventanas emergentes (alert()).

1. Código para Validación del Campo de Texto

const validarTexto = () => {
        let campoTexto = document.getElementById('miCampoTexto').value;
        if (campoTexto === "") {
          alert("El campo no puede estar vacío");
          return false;
        }
        return true;
      };
      
      const formulario = document.getElementById('formulario');
      formulario.addEventListener('submit', (event) => {
        if (!validarTexto()) {
          event.preventDefault();
        }
      });

Explicación:

  • 1. let campoTexto = document.getElementById('miCampoTexto').value;: Obtiene el valor del campo de texto.
  • 2. if (campoTexto === ""): Verifica si el campo está vacío.
  • 3. alert("El campo no puede estar vacío");: Si está vacío, muestra una alerta.
  • 4. return false;: Si está vacío, retorna false para detener la validación.
  • 5. return true;: Si tiene contenido, retorna true.

2. Código para Validación de Checkbox

const validarCheckbox = () => {
  let checkbox = document.getElementById('miCheckbox');
  if (!checkbox.checked) {
    alert("Debe aceptar los términos");
    return false;
  }
  return true;
};

Explicación:

  • 1. let checkbox = document.getElementById('miCheckbox');: Obtiene el estado del checkbox.
  • 2. if (!checkbox.checked): Verifica si el checkbox no está marcado.
  • 3. alert("Debe aceptar los términos");: Muestra una alerta si no está marcado.
  • 4. return false;: Retorna false si no está marcado.
  • 5. return true;: Retorna true si está marcado.

3. Código para Validación de Select (Menú Desplegable)

const validarSelect = () => {
  let select = document.getElementById('miSelect');
  if (select.selectedIndex === 0) {
    alert("Debe seleccionar una opción válida");
    return false;
  }
  return true;
};

Explicación:

  • 1. let select = document.getElementById('miSelect');: Obtiene el menú desplegable con el id miSelect.
  • 2. if (select.selectedIndex === 0): Verifica si no se ha seleccionado una opción válida.
  • 3. alert("Debe seleccionar una opción válida");: Muestra una alerta si no se selecciona una opción válida.
  • 4. return false;: Retorna false si no se selecciona una opción válida.
  • 5. return true;: Retorna true si se selecciona una opción válida.

4. Código para Validación de Número

const validarNumero = () => {
  let numberField = document.getElementById('numberField');
  let numberValue = parseInt(numberField.value, 10);
  if (isNaN(numberValue) || numberValue < 1 || numberValue > 100) {
    alert("El número debe estar entre 1 y 100");
    return false;
  }
  return true;
};

Explicación:

  • 1. let numberField = document.getElementById('numberField');: Obtiene el valor del campo numérico.
  • 2. let numberValue = parseInt(numberField.value, 10);: Convierte el valor a número entero.
  • 3. if (isNaN(numberValue) || numberValue < 1 || numberValue > 100): Verifica si el número no es válido.
  • 4. alert("El número debe estar entre 1 y 100");: Si no es válido, muestra una alerta.
  • 5. return false;: Retorna false si el número no es válido.
  • 6. return true;: Retorna true si el número es válido.

5. Validación de Archivo (JPEG o PNG)

const validarArchivo = () => {
  let fileInput = document.getElementById('fileInput');
  let file = fileInput.files[0];
  if (file) {
    let allowedTypes = ['image/jpeg', 'image/png'];
    if (!allowedTypes.includes(file.type)) {
      alert("Solo se permiten archivos JPEG o PNG");
      return false;
    }
    if (file.size > 2000000) { // 2MB
      alert("El archivo no puede superar los 2MB");
      return false;
    }
  }
  return true;
};

Explicación:

  • 1. let fileInput = document.getElementById('fileInput');: Obtiene el campo de archivo con el id fileInput.
  • 2. let file = fileInput.files[0];: Obtiene el primer archivo seleccionado.
  • 3. if (file): Verifica si se ha seleccionado un archivo.
  • 4. let allowedTypes = ['image/jpeg', 'image/png'];: Define los tipos de archivo permitidos (JPEG y PNG).
  • 5. if (!allowedTypes.includes(file.type)): Verifica si el tipo de archivo no es válido.
  • 6. alert("Solo se permiten archivos JPEG o PNG");: Si el tipo de archivo no es válido, muestra una alerta.
  • 7. if (file.size > 2000000): Verifica si el tamaño del archivo supera los 2MB.
  • 8. alert("El archivo no puede superar los 2MB");: Si el archivo es demasiado grande, muestra una alerta.
  • 9. return true;: Si el archivo es válido, retorna true.

1. Validación General

let formulario = document.getElementById('formulario');
    
    formulario.addEventListener('submit', event => {
      if (!validarTexto() || !validarCheckbox() || !validarSelect() || !validarNumero() || !validarArchivo()) {
        event.preventDefault(); // Detiene el envío si hay errores
      }
    });

Explicación:

  • 1. let formulario = document.getElementById('formulario');: Obtiene el formulario con el id formulario.
  • 2. formulario.addEventListener('submit', event => {...});: Asocia un evento de envío (submit) al formulario.
  • 3. if (!validarTexto() || !validarCheckbox() || ...): Llama a varias funciones de validación. Si alguna retorna false, impide que el formulario se envíe.
  • 4. event.preventDefault();: Evita el envío del formulario cuando hay errores de validación.

Validación General Optimizada

let formulario = document.getElementById('formulario');
    
    let validaciones = [validarTexto, validarCheckbox, validarSelect, validarNumero, validarArchivo];
    
    formulario.addEventListener('submit', event => {
      for (let validar of validaciones) {
        if (!validar()) {
          event.preventDefault(); // Detiene el envío si hay errores
          break;
        }
      }
    });

Explicación:

  • 1. let validaciones = [validarTexto, validarCheckbox, validarSelect, validarNumero, validarArchivo];: Se crea un array con todas las funciones de validación.
  • 2. for (let validar of validaciones): Recorre el array de funciones de validación una por una.
  • 3. if (!validar()): Llama a cada función de validación. Si alguna retorna false, detiene el envío del formulario.
  • 4. event.preventDefault();: Si alguna función falla, se previene el envío del formulario.
  • 5. break;: Sale del bucle en cuanto encuentra la primera validación fallida.
